function glink {
    param(
        [string]$branch
    )
    
    $remoteUrl = git remote get-url origin
    $baseUrl = $remoteUrl -replace '\.git$', ''  # Remove .git suffix if present

    if ($branch) {
        $url = "$baseUrl/tree/$branch"
    } else {
        $url = $baseUrl
    }
    
    Write-Output $url
    Start-Process $url
}

function forward {
    param (
        [int]$port = 3000
    )
    
    $ngrokProcess = Start-Process "powershell" -ArgumentList "ngrok http $port" -PassThru
    
    Start-Sleep -Seconds 3
    
    $ngrokUrl = & "powershell" -Command {
        $ngrokInfo = curl http://localhost:4040/api/tunnels
        $ngrokData = $ngrokInfo | ConvertFrom-Json
        $ngrokData.tunnels[0].public_url
    }

    Start-Process $ngrokUrl
}
